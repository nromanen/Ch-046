create table timeTable (id serial not null, subjects_id integer references subjects(id), pairs_id integer references pairs(id), teachers_id integer references teachers(id), groups_id integer references groups(id), dayofweeks_id integer references dayofweeks(id), oddnes_id integer references oddnes(id), classrooms_id integer references classrooms(id), primary key(id));

-- insert into timetable (subjects_id, pairs_id, teachers_id, groups_id, dayofweeks_id, oddnes_id, classrooms_id) values (2,2,2,2,2,2,95);

select groups_id, groups.name,
subjects_id, subjects.name,
teachers_id,  teachers.lastname, teachers.firstname,
dayofweeks_id, dayofweeks.name,
pairs_id, pairs.name,
oddnes_id, oddnes.name,
classrooms_id, classrooms.name
from timetable
join subjects on subjects.id = subjects_id
join teachers on teachers.id = teachers_id
join pairs on pairs.id = pairs_id
join groups on groups.id = groups_id
join dayofweeks on dayofweeks.id = dayofweeks_id
join oddnes on oddnes.id = oddnes_id
join classrooms on classrooms.id = classrooms_id
where timetable.id = 1;
